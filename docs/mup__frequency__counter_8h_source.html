68 unsigned long getResetpIrqCount(uint8_t irqno) {
69  unsigned long count = (unsigned long)-1;
70  noInterrupts();
71  if (irqno < USTD_MAX_PIRQS) {
72  count = pirqcounter[irqno];
73  pirqcounter[irqno] = 0;

79 double frequencyMultiplicator = 1000000.0;
80 double getResetpIrqFrequency(uint8_t irqno, unsigned long minDtUs = 50) {
81  double frequency = 0.0;
82  noInterrupts();
83  if (irqno < USTD_MAX_PIRQS) {
84  unsigned long count = pirqcounter[irqno];
85  unsigned long dt = timeDiff(pbeginIrqTimer[irqno], plastIrqTimer[irqno]);
86  if (dt > minDtUs) { // Ignore small Irq flukes
87  frequency = (count * frequencyMultiplicator) / dt; // = count/2.0*1000.0000 uS / dt;
88  // no. of waves (count/2) / dt.
89  }
90  pbeginIrqTimer[irqno] = 0;
91  pirqcounter[irqno] = 0;
92  plastIrqTimer[irqno] = 0;
93  }
94  interrupts();
95  return frequency;
96 }
