20void G_INT_ATTR ustd_sw_irq_master(uint8_t irqno) {
21 unsigned long curr = millis();
22 noInterrupts();
23 if (pSwDebounceMs[irqno]) {
24 if (timeDiff(pSwLastIrq[irqno], curr) < pSwDebounceMs[irqno]) {
25 interrupts();
26 return;
27 }
28 }
29 ++pSwIrqCounter[irqno];
30 pSwLastIrq[irqno] = curr;
31 interrupts();
32}

240 void begin(Scheduler *_pSched) {
241 // clang-format off
265 // clang-format on
266 pSched = _pSched;
267
268 pinMode(port, INPUT_PULLUP);
269 setMode(mode);
270
271 if (interruptIndex >= 0 && interruptIndex < USTD_SW_MAX_IRQS) {
272 ipin = digitalPinToInterrupt(port);
273 switch (mode) {
274 case Mode::Falling:
275 attachInterrupt(ipin, ustd_sw_irq_table[interruptIndex], FALLING);
276 break;
277 case Mode::Rising:
278 attachInterrupt(ipin, ustd_sw_irq_table[interruptIndex], RISING);
279 break;
280 default:
281 attachInterrupt(ipin, ustd_sw_irq_table[interruptIndex], CHANGE);
282 break;
283 }
284 pSwDebounceMs[interruptIndex] = debounceTimeMs;
285 useInterrupt = true;
286 }

294 this->subsMsg(topic, msg, originator);

334 void publishLogicalState(bool lState) {
335 String textState;
336 if (lState == true)
337 textState = "on";
338 else
339 textState = "off";
340 switch (mode) {
341 case Mode::Default:
342 case Mode::Flipflop:
343 case Mode::Timer:
344 pSched->publish(name + "/switch/state", textState);
345 if (customTopic != "")
346 pSched->publish(customTopic, textState);
347 break;
348 case Mode::Rising:
349 if (lState == true) {
350 pSched->publish(name + "/switch/state", "trigger");
351 if (customTopic != "")
352 pSched->publish(customTopic, "trigger");
353 }
354 break;
355 case Mode::Falling:
356 if (lState == false) {
357 pSched->publish(name + "/switch/state", "trigger");
358 if (customTopic != "")
359 pSched->publish(customTopic, "trigger");
360 }
361 break;
362 case Mode::Duration:
363 if (lState == true) {
364 startEvent = millis();
365 } else {
366 if (startEvent != (unsigned long)-1) {
367 unsigned long dt = timeDiff(startEvent, millis());
368 char msg[32];
369 sprintf(msg, "%ld", dt);
370 pSched->publish(name + "/switch/duration", msg);
371 if (dt < durations[0]) {
372 pSched->publish(name + "/switch/shortpress", "trigger");
373 } else if (dt < durations[1]) {
374 pSched->publish(name + "/switch/longpress", "trigger");
375 } else {
376 pSched->publish(name + "/switch/verylongpress", "trigger");
377 }
378 }
379 }
380 break;
381 }

384 void decodeLogicalState(bool physicalState) {
385 switch (mode) {
386 case Mode::Default:
387 case Mode::Rising:
388 case Mode::Falling:
389 case Mode::Duration:
390 setLogicalState(physicalState);
391 break;
392 case Mode::Flipflop:
393 if (physicalState == false) {
394 flipflop = !flipflop;
395 setLogicalState(flipflop);
396 }
397 break;
398 case Mode::Timer:
399 if (physicalState == false) {
400 activeTimer = millis();
401 } else {
402 setLogicalState(true);
403 }
404 break;
405 }

408 void setPhysicalState(bool newState, bool override) {
409 if (mode != Mode::Timer) {
410 activeTimer = 0;
411 }
412 if (override) {
413 overriddenPhysicalState = physicalState;
414 overridePhysicalActive = true;
415 if (newState != physicalState) {
416 physicalState = newState;
417 decodeLogicalState(newState);
418 }
419 } else {
420 if (overridePhysicalActive && newState != overriddenPhysicalState) {
421 overridePhysicalActive = false;
422 }
423 if (overridePhysicalActive)
424 return;
425 if (newState != physicalState || mode == Mode::Falling || mode == Mode::Rising) {
426 if (timeDiff(lastChangeMs, millis()) > debounceTimeMs || useInterrupt) {
427 lastChangeMs = millis();
428 physicalState = newState;
429 decodeLogicalState(physicalState);
430 }
431 }

435 void readState() {
436 if (useInterrupt) {
437 unsigned long count = getSwResetIrqCount(interruptIndex);
438 int curstate = digitalRead(port);
439 char msg[32];
440 if (count) {
441 sprintf(msg, "%ld", count);
442 pSched->publish(name + "/switch/irqcount/0", msg);
443 if (curstate == HIGH)
444 curstate = true;
445 else
446 curstate = false;
447 switch (mode) {
448 case Mode::Rising:
449 for (unsigned long i = 0; i < count; i++) {
450 if (activeLogic) {
451 setPhysicalState(false, false);
452 setPhysicalState(true, false);
453 } else {
454 setPhysicalState(true, false);
455 setPhysicalState(false, false);
456 }
457 }
458 break;
459 case Mode::Falling:
460 for (unsigned long i = 0; i < count; i++) {
461 if (activeLogic) {
462 setPhysicalState(true, false);
463 setPhysicalState(false, false);
464 } else {
465 setPhysicalState(false, false);
466 setPhysicalState(true, false);
467 }
468 }
469 break;
470 default:
471 bool iState = ((count % 2) == 0);
472 if (curstate)
473 iState = !iState;
474 for (unsigned long i = 0; i < count; i++) {
475 if (activeLogic) {
476 setPhysicalState(iState, false);
477 } else {
478 setPhysicalState(!iState, false);
479 }
480 iState = !iState;
481 }
482 break;
483 }
484 }
485 } else {
486 int newstate = digitalRead(port);
487 if (newstate == HIGH)
488 newstate = true;
489 else
490 newstate = false;
491 if (!activeLogic)
492 newstate = !newstate;
493 setPhysicalState(newstate, false);
494 }
495 }

497 void loop() {
498 readState();
499 if (mode == Mode::Timer && activeTimer) {
500 if (timeDiff(activeTimer, millis()) > timerDuration) {
501 activeTimer = 0;
502 setLogicalState(false);
503 }
504 }

507 void subsMsg(String topic, String msg, String originator) {
508 if (topic == name + "/switch/state/get") {
509 char buf[32];
510 if (logicalState)
511 sprintf(buf, "on");
512 else
513 sprintf(buf, "off");
514 pSched->publish(name + "/switch/state", buf);
515 }
516 if (topic == name + "/switch/physicalstate/get") {
517 char buf[32];
518 if (physicalState)
519 sprintf(buf, "on");
520 else
521 sprintf(buf, "off");
522 pSched->publish(name + "/switch/physicalstate", buf);
523 }
524 if (topic == name + "/switch/mode/set") {
525 char buf[32];
526 memset(buf, 0, 32);
527 strncpy(buf, msg.c_str(), 31);
528 char *p = strchr(buf, ' ');
529 char *p2 = nullptr;
530 if (p) {
531 *p = 0;
532 ++p;
533 p2 = strchr(p, ',');
534 if (p2) {
535 *p2 = 0;
536 ++p2;
537 }
538 }
539 if (!strcmp(buf, "default")) {
540 setMode(Mode::Default);
541 } else if (!strcmp(buf, "rising")) {
542 setMode(Mode::Rising);
543 } else if (!strcmp(buf, "falling")) {
544 setMode(Mode::Falling);
545 } else if (!strcmp(buf, "flipflop")) {
546 setMode(Mode::Flipflop);
547 } else if (!strcmp(buf, "timer")) {
548 unsigned long dur = 1000;
549 if (p)
550 dur = atol(p);
551 setMode(Mode::Timer, dur);
552 } else if (!strcmp(buf, "duration")) {
553 durations[0] = 3000;
554 durations[1] = 30000;
555 if (p) {
556 durations[0] = atol(p);
557 if (p2)
558 durations[1] = atol(p2);
559 }
560 if (durations[0] > durations[1]) {
561 durations[1] = (unsigned long)-1;
562 }
563 setMode(Mode::Duration);
564 }
565 }
566 if (topic == name + "/switch/set") {
567 char buf[32];
568 memset(buf, 0, 32);
569 strncpy(buf, msg.c_str(), 31);
570 if (!strcmp(buf, "on") || !strcmp(buf, "true")) {
571 // setPhysicalState(true, true);
572 setLogicalState(true);
573 }
574 if (!strcmp(buf, "off") || !strcmp(buf, "false")) {
575 // setPhysicalState(false, true);
576 setLogicalState(false);
577 }
578 if (!strcmp(buf, "toggle")) {
579 setToggle();
580 }
581 if (!strcmp(buf, "pulse")) {
582 setPulse();
583 }
584 }
585 if (topic == name + "/switch/debounce/get") {
586 char buf[32];
587 sprintf(buf, "%ld", debounceTimeMs);
588 pSched->publish(name + "/debounce", buf);
589 }
590 if (topic == name + "/switch/debounce/set") {
591 long dbt = atol(msg.c_str());
592 setDebounce(dbt);
593 }
594 if (topic == "mqtt/state") {
595 if (mode == Mode::Default || mode == Mode::Flipflop) {
596 if (msg == "connected") {
597 publishLogicalState(logicalState);
598 }
599 }
600 }
