20void G_INT_ATTR ustd_sw_irq_master(uint8_t irqno) {
21 unsigned long curr = millis();
22 noInterrupts();
23 if (pSwDebounceMs[irqno]) {
24 if (timeDiff(pSwLastIrq[irqno], curr) < pSwDebounceMs[irqno]) {
25 interrupts();
26 return;
27 }
28 }
29 ++pSwIrqCounter[irqno];
30 pSwLastIrq[irqno] = curr;
31 interrupts();
32}

375 textState = "off";
376 switch (mode) {
377 case Mode::Default:
378 case Mode::Flipflop:
379 case Mode::Timer:

381 if (customTopic != "")

384 case Mode::Rising:
385 if (lState == true) {
386 pSched->publish(name + "/switch/state", "trigger");

397 break;
398 case Mode::Duration:

400 startEvent = millis();

403 unsigned long dt = timeDiff(startEvent, millis());

406 pSched->publish(name + "/switch/duration", msg);

420 void decodeLogicalState(bool physicalState) {
421 switch (mode) {
422 case Mode::Default:
423 case Mode::Rising:
424 case Mode::Falling:
425 case Mode::Duration:
426 setLogicalState(physicalState);
427 break;
428 case Mode::Flipflop:
429 if (physicalState == false) {
430 flipflop = !flipflop;
431 setLogicalState(flipflop);
432 }
433 break;
434 case Mode::Timer:
435 if (physicalState == false) {
436 activeTimer = millis();
437 } else {
438 setLogicalState(true);
439 }
440 break;
441 }

444 void setPhysicalState(bool newState, bool override) {
445 if (mode != Mode::Timer) {
446 activeTimer = 0;
447 }
448 if (override) {
449 overriddenPhysicalState = physicalState;
450 overridePhysicalActive = true;
451 if (newState != physicalState) {
452 physicalState = newState;
453 decodeLogicalState(newState);
454 }
455 } else {
456 if (overridePhysicalActive && newState != overriddenPhysicalState) {
457 overridePhysicalActive = false;
458 }
459 if (overridePhysicalActive)
460 return;
461 if (newState != physicalState || mode == Mode::Falling || mode == Mode::Rising) {
462 if (timeDiff(lastChangeMs, millis()) > debounceTimeMs || useInterrupt) {
463 lastChangeMs = millis();
464 physicalState = newState;
465 decodeLogicalState(physicalState);
466 }
467 }

471 void readState() {
472 if (useInterrupt) {
473 unsigned long count = getSwResetIrqCount(interruptIndex);
474 int curstate = digitalRead(port);
475 char msg[32];
476 if (count) {
477 sprintf(msg, "%ld", count);
478 pSched->publish(name + "/switch/irqcount/0", msg);
479 if (curstate == HIGH)
480 curstate = true;
481 else
482 curstate = false;
483 switch (mode) {
484 case Mode::Rising:
485 for (unsigned long i = 0; i < count; i++) {
486 if (activeLogic) {
487 setPhysicalState(false, false);
488 setPhysicalState(true, false);
489 } else {
490 setPhysicalState(true, false);
491 setPhysicalState(false, false);
492 }
493 }
494 break;
495 case Mode::Falling:
496 for (unsigned long i = 0; i < count; i++) {
497 if (activeLogic) {
498 setPhysicalState(true, false);
499 setPhysicalState(false, false);
500 } else {
501 setPhysicalState(false, false);
502 setPhysicalState(true, false);
503 }
504 }
505 break;
506 default:
507 bool iState = ((count % 2) == 0);
508 if (curstate)
509 iState = !iState;
510 for (unsigned long i = 0; i < count; i++) {
511 if (activeLogic) {
512 setPhysicalState(iState, false);
513 } else {
514 setPhysicalState(!iState, false);
515 }
516 iState = !iState;
517 }
518 break;
519 }
520 }
521 } else {
522 int newstate = digitalRead(port);
523 if (newstate == HIGH)
524 newstate = true;
525 else
526 newstate = false;
527 if (!activeLogic)
528 newstate = !newstate;
529 setPhysicalState(newstate, false);
530 }
531 }

534 readState();

537 activeTimer = 0;

543 void subsMsg(String topic, String msg, String originator) {
544 if (topic == name + "/switch/state/get") {
545 char buf[32];
546 if (logicalState)
547 sprintf(buf, "on");
548 else
549 sprintf(buf, "off");
550 pSched->publish(name + "/switch/state", buf);
551 } else if (topic == name + "/switch/counter/get") {
552 publishCounter();
553 } else if (topic == name + "/switch/physicalstate/get") {
554 char buf[32];
555 if (physicalState)
556 sprintf(buf, "on");
557 else
558 sprintf(buf, "off");
559 pSched->publish(name + "/switch/physicalstate", buf);
560 } else if (topic == name + "/switch/mode/set") {
561 char buf[32];
562 memset(buf, 0, 32);
563 strncpy(buf, msg.c_str(), 31);
564 char *p = strchr(buf, ' ');
565 char *p2 = nullptr;
566 if (p) {
567 *p = 0;
568 ++p;
569 p2 = strchr(p, ',');
570 if (p2) {
571 *p2 = 0;
572 ++p2;
573 }
574 }
575 if (!strcmp(buf, "default")) {
576 setMode(Mode::Default);
577 } else if (!strcmp(buf, "rising")) {
578 setMode(Mode::Rising);
579 } else if (!strcmp(buf, "falling")) {
580 setMode(Mode::Falling);
581 } else if (!strcmp(buf, "flipflop")) {
582 setMode(Mode::Flipflop);
583 } else if (!strcmp(buf, "timer")) {
584 unsigned long dur = 1000;
585 if (p)
586 dur = atol(p);
587 setMode(Mode::Timer, dur);
588 } else if (!strcmp(buf, "duration")) {
589 durations[0] = 3000;
590 durations[1] = 30000;
591 if (p) {
592 durations[0] = atol(p);
593 if (p2)
594 durations[1] = atol(p2);
595 }
596 if (durations[0] > durations[1]) {
597 durations[1] = (unsigned long)-1;
598 }
599 setMode(Mode::Duration);
600 }
601 } else if (topic == name + "/switch/set") {
602 char buf[32];
603 memset(buf, 0, 32);
604 strncpy(buf, msg.c_str(), 31);
605 if (!strcmp(buf, "on") || !strcmp(buf, "true")) {
606 // setPhysicalState(true, true);
607 setLogicalState(true);
608 }
609 if (!strcmp(buf, "off") || !strcmp(buf, "false")) {
610 // setPhysicalState(false, true);
611 setLogicalState(false);
612 }
613 if (!strcmp(buf, "toggle")) {
614 setToggle();
615 }
616 if (!strcmp(buf, "pulse")) {
617 setPulse();
618 }
619 } else if (topic == name + "/switch/debounce/get") {
620 char buf[32];
621 sprintf(buf, "%ld", debounceTimeMs);
622 pSched->publish(name + "/debounce", buf);
623 } else if (topic == name + "/switch/debounce/set") {
624 long dbt = atol(msg.c_str());
625 setDebounce(dbt);
626 } else if (topic == "mqtt/state") {
627 if (mode == Mode::Default || mode == Mode::Flipflop) {
628 if (msg == "connected") {
629 publishLogicalState(logicalState);
630 }
631 }
632 } else if (topic == name + "/switch/counter/start") {
633 activateCounter(true);
634 } else if (topic == name + "/switch/counter/stop") {
635 activateCounter(false);
636 }
637 };
