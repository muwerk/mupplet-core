132 static bool calculateSunRiseSet(int year, int month, int day, double lat, double lon,
133 int localOffset, int daylightSavings, bool bRising,
134 double *pSunTime) {
141 // 1. first calculate the day of the year
142 const double ZENITH = 90.0 + 50.0 / 60.0;
143 double N1 = floor(275.0 * month / 9.0);
144 double N2 = floor((month + 9.0) / 12.0);
145 double N3 = (1.0 + floor((year - 4 * floor(year / 4.0) + 2.0) / 3.0));
146 double N = N1 - (N2 * N3) + day - 30.0;
147
148 // 2. convert the longitude to hour value and calculate an approximate time
149 double lonHour = lon / 15.0;
150 double t;
151 if (bRising)
152 t = N + ((6 - lonHour) / 24); // if rising time is desired:
153 else
154 t = N + ((18 - lonHour) / 24); // if setting time is desired:
155
156 // 3. calculate the Sun's mean anomaly
157 double M = (0.9856 * t) - 3.289;
158
159 // 4. calculate the Sun's true longitude
160 double L =
161 fmod(M + (1.916 * sin(C_D2R * M)) + (0.020 * sin(2 * C_D2R * M)) + 282.634, 360.0);
162
163 // 5a. calculate the Sun's right ascension
164 double RA = fmod(C_R2D * atan(0.91764 * tan(C_D2R * L)), 360.0);
165
166 // 5b. right ascension value needs to be in the same quadrant as L
167 double Lquadrant = floor(L / 90.0) * 90.0;
168 double RAquadrant = floor(RA / 90.0) * 90.0;
169 RA = RA + (Lquadrant - RAquadrant);
170
171 // 5c. right ascension value needs to be converted into hours
172 RA = RA / 15.0;
173
174 // 6. calculate the Sun's declination
175 double sinDec = 0.39782 * sin(C_D2R * L);
176 double cosDec = cos(asin(sinDec));
177
178 // 7a. calculate the Sun's local hour angle
179 // double cosH = (sin(C_D2R * ZENITH) - (sinDec * sin(C_D2R * lat))) /
180 // (cosDec * cos(C_D2R * lat));
181 double cosH =
182 (cos(C_D2R * ZENITH) - (sinDec * sin(lat * C_D2R))) / (cosDec * cos(C_D2R * lat));
183
184 if (bRising) {
185 if (cosH > 1.0) { // the sun never rises on this location (on the specified date)
186 *pSunTime = -1.0;
187 return false;
188 }
189 } else {
190 if (cosH < -1.0) { // the sun never sets on this location (on the specified date)
191 *pSunTime = -1.0;
192 return false;
193 }
194 }
195
196 // 7b. finish calculating H and convert into hours
197 double H;
198 if (bRising)
199 H = 360.0 - C_R2D * acos(cosH); // if if rising time is desired:
200 else
201 H = C_R2D * acos(cosH); // if setting time is desired:
202 H = H / 15.0;
203
204 // 8. calculate local mean time of rising/setting
205 double T = H + RA - (0.06571 * t) - 6.622;
206
207 // 9. adjust back to UTC
208 double UT = fmod(T - lonHour, 24.0);
209
210 // 10. convert UT value to local time zone of latitude/longitude
211 *pSunTime = UT + localOffset + daylightSavings;
212 return true;
213 }
