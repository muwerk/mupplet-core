301  if (mode == Mode::Passive)
302  return;
303  unsigned long period = (millis() + uPhase) % (2 * interval);
304  if (mode == Mode::Pulse) {

311  }
312  if (mode == Mode::Blink) {
313  if (period < oPeriod) {
314  set(false, true);
315  } else {

317  set(true, true);
318  }
319  }
320  }
321  if (mode == Mode::Wave) {
322  unsigned long period = (millis() + uPhase) % (2 * interval);
323  double br = 0.0;
324  if (period < interval) {
325  br = (double)period / (double)interval;
326  } else {
327  br = (double)(2 * interval - period) / (double)interval;
328  }
329  brightness(br, true);
330  }
331  if (mode == Mode::Pattern) {
332  if (period < oPeriod) {
333  if (patternPointer < pattern.length()) {
334  char c = pattern[patternPointer];
335  if (c == 'r') {
336  patternPointer = 0;
337  c = pattern[patternPointer];
338  }
339  if (c == '+') {
340  set(true, true);
341  }
342  if (c == '-') {
343  set(false, true);
344  }
345  if (c >= '0' && c <= '9') {
346  double br = (double)(c - '0') * 0.1111;
347  brightness(br, true);
348  }
349  ++patternPointer;
350  } else {
351  patternPointer = 0;
352  set(false, true);
353  setMode(Mode::Passive);
354  }
355  }
