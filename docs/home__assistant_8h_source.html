192 void begin(Scheduler *_pSched, bool initialAutodiscovery = false) {
193 pSched = _pSched;
194
195 // initialize configuration
196#ifdef USTD_FEATURE_FILESYSTEM
197 autodiscovery = config.readBool("ha/autodiscovery", initialAutodiscovery);
198 if ((deviceId = config.readString("net/deviceid")) == "") {
199 // initialize device id to mac address
200 deviceId = WiFi.macAddress();
201 deviceId.replace(":", "");
202 }
203#else
204 autodiscovery = initialAutodiscovery;
205 deviceId = WiFi.macAddress();
206 deviceId.replace(":", "");
207#endif
208
209 // try to initialze some stuff
210 addAttributes("device");
211
212 // react to network state changes
213 pSched->subscribe(tID, "mqtt/config", [this](String topic, String msg, String orig) {
214 this->onMqttConfig(topic, msg, orig);
215 });
216 pSched->subscribe(tID, "mqtt/state", [this](String topic, String msg, String orig) {
217 this->onMqttState(topic, msg, orig);
218 });
219 pSched->subscribe(tID, "net/network", [this](String topic, String msg, String orig) {
220 this->onNetNetwork(topic, msg, orig);
221 });
222 pSched->subscribe(tID, "net/rssi", [this](String topic, String msg, String orig) {
223 this->onNetRssi(topic, msg, orig);
224 });
225
226 // react to commands
227 pSched->subscribe(tID, "ha/state/#", [this](String topic, String msg, String originator) {
228 this->onCommand(topic.substring(11), msg);
229 });
230
231 // request current state
232 pSched->publish("net/network/get");
233 pSched->publish("mqtt/state/get");
234
235 // publish out current state
236 publishState();
237 }

270 void addAttributes(String attribGroup, String manufacturer = "", String model = "",
271 String version = "") {
272
273 for (unsigned int i = 0; i < attribGroups.length(); i++) {
274 if (attribGroup == attribGroups[i].name) {
275 // insert only unique attribute groups
276 return;
277 }
278 }
279
280 Attributes att = {};
281 String *man = manufacturer.length() ? &manufacturer : &deviceManufacturer;
282 String *mod = model.length() ? &model : &deviceModel;
283 String *ver = version.length() ? &version : &deviceVersion;

285 att.name = (char *)malloc(attribGroup.length() + man->length() + mod->length() +
286 ver->length() + 4);
287 if (att.name) {

649 void onCommand(String topic, String msg) {
650 if (topic == "get") {
651 publishState();
652 } else if (topic == "set") {
653 msg.trim();
654 msg.toLowerCase();
655 if (msg == "on" || msg == "true") {
656 setAutoDiscovery(true);
657 } else if (msg == "off" || msg == "false") {
658 setAutoDiscovery(false);

805 void publishConfig(Entity &entity) {
806 String name = getEntityName(entity);
807 String key = getEntityKey(entity);
808 String topic = getEntityTopic(entity);
809
810 if (entity.channel == -1) {
811 publishConfig(entity, name, key, topic);
812 } else if (entity.channel < -1) {
813 for (int i = 0; i < abs(entity.channel); i++) {
814 publishConfig(entity, name + "." + i, key + "_" + i, topic + "/" + i);
815 }
816 } else {
817 int i = entity.channel;
818 publishConfig(entity, name + "." + i, key + "_" + i, topic + "/" + i);
819 }

822 void publishConfig(Entity &entity, String name, String key, String topic) {
823 JSONVar msg;
824 msg["~"] = pathPrefix + "/";
825 msg["name"] = hostName + " " + name;
826 msg["uniq_id"] = key;
827 msg["avty_t"] = "~mqtt/state";
828 msg["pl_avail"] = "connected";
829 msg["pl_not_avail"] = lastWillMessage;
830 msg["json_attr_t"] = "~" + haTopicAttrib + (*entity.attribs ? entity.attribs : "device");
831 if (*entity.dev_cla) {
832 msg["dev_cla"] = entity.dev_cla;
833 }
834 if (*entity.icon) {
835 msg["ic"] = entity.icon;
836 }
837 switch (entity.type) {
838 // case DeviceType::Cover:
839 // publishCoverConfig(msg, entity, topic);
840 // break;
841 case DeviceType::Light:
842 case DeviceType::LightDim:
843 case DeviceType::LightRGB:
844 case DeviceType::LightRGBW:
845 case DeviceType::LightRGBWW:
846 case DeviceType::LightWW:
847 publishLightConfig(msg, entity, topic);
848 break;
849 case DeviceType::Sensor:
850 case DeviceType::BinarySensor:
851 publishSensorConfig(msg, entity, topic);
852 break;
853 case DeviceType::Switch:
854 publishSwitchConfig(msg, entity, topic);
855 break;
856 default:
857 break;
858 }

873 msg["on_cmd_type"] = "brightness";
874 }
875
876 flushDeviceConfig(entity.type, msg);

972 static int WifiGetRssiAsQuality(int rssi) {
973 int quality = 0;
974
975 if (rssi <= -100) {
976 quality = 0;
977 } else if (rssi >= -50) {
978 quality = 100;
979 } else {
980 quality = 2 * (rssi + 100);
981 }
982 return quality;
