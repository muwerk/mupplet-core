360  void loop() {
361  if (mode == Mode::Passive)
362  return;
363  unsigned long period = (millis() + uPhase) % (2 * interval);
364  if (mode == Mode::Pulse) {
365  if (millis() - startPulse < interval) {
366  set(true, true);
367  } else {
368  set(false, true);
369  setMode(Mode::Passive);
370  }
371  }
372  if (mode == Mode::Blink) {
373  if (period < oPeriod) {
374  set(false, true);
375  } else {
376  if (period > interval && oPeriod < interval) {
377  set(true, true);
378  }
379  }
380  }
381  if (mode == Mode::Wave) {
382  unsigned long period = (millis() + uPhase) % (2 * interval);
383  double br = 0.0;
384  if (period < interval) {
385  br = (double)period / (double)interval;
386  } else {
387  br = (double)(2 * interval - period) / (double)interval;
388  }
389  br = br * (maxWaveBrightness - minWaveBrightness) + minWaveBrightness;
390  brightness(br, true);
391  }
392  if (mode == Mode::Pattern) {
393  if (period < oPeriod) {
394  if (patternPointer < pattern.length()) {
395  char c = pattern[patternPointer];
396  if (c == 'r') {
397  patternPointer = 0;
398  c = pattern[patternPointer];
399  }
400  if (c == '+') {
401  set(true, true);
402  }
403  if (c == '-') {
404  set(false, true);
405  }
406  if (c >= '0' && c <= '9') {
407  double br = (double)(c - '0') * 0.1111;
408  brightness(br, true);
409  }
410  ++patternPointer;
411  } else {
412  patternPointer = 0;
413  set(false, true);
414  setMode(Mode::Passive);
415  }
416  }
417  }
418  oPeriod = period;
419  }
