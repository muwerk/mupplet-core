255String utf8ToLatin(String utf8string, char invalid_char = '_') {
272 String latin = "";
273 unsigned char c, nc;
274 for (unsigned int i = 0; i < utf8string.length(); i++) {
275 c = utf8string[i];
276 if ((c & 0x80) == 0) { // 1 byte utf8
277 latin += (char)c;
278 continue;
279 } else {
280 if ((c & 0b11100000) == 0b11000000) { // 2 byte utf8
281 if (i < utf8string.length() - 1) {
282 nc = utf8string[i + 1];
283 switch (c) {
284 case 0xc2:
285 latin += (char)nc;
286 break;
287 case 0xc3:
288 nc += 0x40;
289 latin += (char)nc;
290 break;
291 default:
292 latin += (char)invalid_char;
293 break;
294 }
295 i += 1;
296 continue;
297 } else { // damaged utf8
298 latin += (char)invalid_char;
299 return latin;
300 }
301 } else {
302 if ((c & 0b11110000) == 0b11100000) { // 3 byte utf8
303 i += 2;
304 latin += (char)invalid_char;
305 continue;
306 } else {
307 if ((c & 0b11111000) == 0b11110000) { // 4 byte utf8
308 i += 3;
309 latin += (char)invalid_char;
310 continue;
311 } else { // damaged utf8
312 latin += (char)invalid_char;
313 return latin; // we can't continue parsing
314 }
315 }
316 }
317 }
318 }
